/**
 * Defines a block with simple message.
 *
 * @Block(
 *   id = "simple_message",
 *   admin_label = @Translation("Simple Message"),
 * )
 */
class SimpleMessageBlock extends BlockBase {
  /**
   * {@inheritdoc}
   */
  public function build() {
    return [
      'message' => [
        '#markup' => $this->t('Just a simple message.'),
      ],
    ];
  }
}
